\begin{recipe}{Pikanter Eintopf aus Vollreis und Linsen}{4 Portionen}{1 h}

\ingredient[1]{Kaffeelöffel}{\textbf{Kurkuma}, gemahlen}
\ingredient[1]{Kaffeelöffel}{\textbf{Kreuzkümmel}, gemahlen}
\ingredient[1]{Kaffeelöffel}{\textbf{Currypulver}}
In einem grossen Topf mit Öl leicht dämpfen
\ingredient[1]{}{\textbf{Zwiebel}, fein gehackt}
\ingredient[3]{Zehen}{\textbf{Knoblauch}, fein gehackt}
\ingredient[1-3]{}{\textbf{Chilis}, fein gehackt}
\ingredient[2]{Kaffeelöffel}{\textbf{frischer Ingwer}, gerieben}
Dazugeben und mitdämpfen
\ingredient[225]{g}{\textbf{grüne Linsen}}
\ingredient[225]{g}{\textbf{Vollreis}}
\ingredient[1]{Kaffeelöffel}{\textbf{Salz}}
Kurz mitbraten, dann mit Wasser gut zugedeckt für 30 Min. (15 im Dampfkochtopf) köcheln lassen
\ingredient[4]{}{\textbf{hartgekochte Eier}, gewürfelt}
\ingredient[4]{}{\textbf{Bananen}, in Scheiben}
\ingredient[100]{g}{\textbf{Mandeln}, geröstet}
Mit Eier, Bananen und Mandeln garnieren

\end{recipe}
