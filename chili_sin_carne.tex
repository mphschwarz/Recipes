\begin{recipe}{Chili sin Carne}{4 Portionen (max. 2.5x)}{50 Minuten}

\ingredient[1]{}{\textbf{Zwiebel} fein gehackt}
\ingredient[2]{Zehen}{\textbf{Knoblauch} fein gehackt}
\ingredient[1]{}{\textbf{Chilischote} entkernt, fein gehackt}
In Öl andünsten
\ingredient[140]{g}{\textbf{Karotten} fein gewürfelt}
\ingredient[200]{g}{\textbf{Peperoni} gewürfelt}
\ingredient[2]{Esslöffel}{\textbf{Tomatenpüree}}
Beigeben und andünsten ($\sim$3 min.)
\ingredient[400]{g}{\textbf{gehackte geschälte Tomaten}}
Beigeben, etwas köcheln lassen
\ingredient[4]{dl}{\textbf{Gemüsebouillon}}
\ingredient[1]{Teelöffel}{\textbf{Salz}}
\ingredient[\fr1 2]{Teelöffel}{\textbf{Pfeffer}}
Mit Bouillon ablöscen und würzen
\ingredient[30]{g}{\textbf{dunkle Schokolade}}
Beigeben und köcheln lassen bis die Schokolade geschmolzen ist
\ingredient[300]{g}{\textbf{Kidneybohnen}}
\ingredient[250]{g}{\textbf{Borlottibohnen}}
Beide Bohnensorten in ein Sieb abgiessen, abspülen und abtropfen lassen, 
zum Chili geben und bei kleiner Hitze ($\sim$15 min.) zugedeckt köcheln lassen
\ingredient[230]{g}{\textbf{Quorn}}
Beigeben und köcheln lassen ($\sim$10 min.)
\ingredient[1]{Bund}{\textbf{Petersilie}, abgezupft}
Zum Chili servieren

\end{recipe}
