\begin{recipe}{Weisses Geflügelcurry}{4 Portionen}{50 Minuten}

\ingredient[2]{Esslöffel}{\textbf{Bratbutter}}
In einer weiten, beschichteten Bratpfanne warm werden lassen.
\ingredient[2]{}{\textbf{Zwiebeln}, grob gehackt}
\ingredient[50]{g}{\textbf{geschälte Mandeln}}
\ingredient[50]{g}{\textbf{Cashew-Nüsse}}
\ingredient[2]{Zehen}{\textbf{Knoblauch}, in Stücken}
\ingredient[3]{cm}{\textbf{Ingwer}, in Scheiben}
\ingredient[3]{}{\textbf{grüne Chili}, entkernt, in Stücken}
\ingredient[4]{}{\textbf{grüne Kardamomkapseln}, nur ausgelöste Samen}
\ingredient[1]{}{\textbf{Lorbeerblat}}
\ingredient[1]{}{\textbf{Nelke}}
\ingredient[1 \fr1 2]{Teelöffel}{\textbf{Mohnsamen}}
\ingredient[\fr1 2]{Teelöffel}{\textbf{Salz}}
Alles 5 Min. dämpfen.
\ingredient[4]{dl}{\textbf{Wasser}}
Dazugiessen und 15 Min. köcheln lassen.
Lorbeerblatt entfernen und in einem Messbecher pürieren.
\ingredient[1]{Esslöffel}{\textbf{Bratbutter}}
\ingredient[4]{}{\textbf{Pouletbrüstli}, in 4 cm grossen Würfeln}
Beidseitig je 2 Min. anbraten.
\ingredient[\fr3 4]{Teelöffel}{\textbf{Salz}}
Salzen
\ingredient[180]{g}{\textbf{Yogurt}}
Mit beiseite gestellter Sauce mischen und 7 Min. köcheln lassen.

\end{recipe}
