\begin{recipe}{Zitronenrisotto}{}

\ingredient[1]{}{\textbf{Schalotte} fein gehackt}
\ingredient[1]{Zehen}{\textbf{Knoblauch} gepresst}
In Öl andünsten
\ingredient[350]{g}{\textbf{Risottoreis}}
Beigeben und unter rühren dünsten bis glasig
\ingredient[1]{}{\textbf{Zitrone} abgeriebene Schale}
\ingredient[1]{Zweig}{\textbf{Thymian}}
Beigeben
\ingredient[1]{dl}{\textbf{Prosecco}}
\ingredient[1]{Esslöffel}{\textbf{Zitronensaft}}
Mit Prosecco und Zitronensaft abläschen und vollständig einkochen lassen
\ingredient[1]{l}{\textbf{Gemüsebouillon} heiss}
Unter häufigem Rühren portionenweise dazugiessen so dass der Reis immer knapp bedeckt ist (\~20 min.)
\ingredient[20]{g}{\textbf{Butter}}
\ingredient[50]{g}{\textbf{Parmesan} gerieben}
\ingredient[]{}{\textbf{Salz} nach Bedarf}
\ingredient[]{}{\textbf{Pfeffer} nach Bedarf}
Unter den Reis mischen
\ingredient[1]{Esslöffel}{\textbf{Butter}}
\ingredient[1]{}{\textbf{Bundzwiebel} in feinen Ringen}
\ingredient[40]{g}{\textbf{getrocknete Morcheln} eingeweicht, gewaschen, halbiert}
Morcheln und Zwiebel in flüssiger Butter dämpfen (\~10 min.)
\ingredient[\fr1 2]{Teelöffel}{\textbf{Salz}}
Würzen

\end{recipe}
